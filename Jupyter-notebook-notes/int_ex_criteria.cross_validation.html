from itertools import combinations as combi #     
import random as rnd
def CCV_ind(N,k,seed=None,shuffle=True):
    """       N   k """
    R=set(range(N))
    mc = combi(R,k) #          R  k
    lst = [(list(R-set(x)),list(x)) for x in mc] #       :
                                        #           
    if shuffle: #     
        rnd.seed(seed)
        rnd.shuffle(lst)
        return lst
    else:
        return lst

def txq(N,t,q,seed=None):
    """   t     N   q """
    lst=list(range(N)) #  
    rnd.seed(seed)
    n=N//q #   
    r=N%q # "" 
    l=[n for i in range(q)] #   
    for i in range(r): #    ""    r
        l[i]+=1
    L=[] #  
    for i in range(t):
        rnd.shuffle(lst) #   
        tlst=lst.copy() #       
        lt=[] #     
        for j in l:
            lq=[] #    
            for k in range(j): #   j   
                lq.append(tlst.pop())
            lt.append(lq)
        L.append(lt)
    return L
print(txq.__name__,txq.__doc__)

def tt_samples(LL):
    """  ,     - ,  \n     ,        \n:\n[[1,2],[3,4],[5,6]] :\n[\n  [[3, 4, 5, 6], [1, 2]],\n  [[1, 2, 5, 6], [3, 4]],\n  [[1, 2, 3, 4], [5, 6]]\n]"""
    L=[] #  
    n=range(len(LL))
    for i in n:
        tl=[]
        for j in n:
            if j!=i:
                tl.extend(LL[j])
        L.append([tl,LL[i]])
    return L
print(tt_samples.__name__,tt_samples.__doc__)

def rc_samples(N,K,seed=None):
    """    K - ,    -        ,       """
    rnd.seed(seed)
    lst=list(range(N))
    L=[]
    for i in range(K):
        tlst=lst.copy()
        tl=[]
        rnd.shuffle(tlst)
        for i in range(rnd.randint(1,N-1)):
            tl.append(tlst.pop())
        L.append([tlst,tl])
    return L
print(':')
for g in rc_samples(10,10):
    print(g)

def bootstrap(N,K,seed=None):
    """   K   N ,    """
    rnd.seed(seed)
    lst=list(range(N))
    return [[lst[rnd.randrange(0,N)] for i in range(N)] for j in range(K)]

print(':')
